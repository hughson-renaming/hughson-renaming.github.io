def getWordCounts(ideaList,limit=0):
    drops = """
    a i to the in on we etc you be like just with about our it that and of is has for they at when why my don't its isn't 
    name idea because from who have think not as will are but can no if this what it's don't - me then does 
    """.split(' ')
    words = ideaList.expandOn(' ') 
    wordData = words[words != '']
    wordData = wordData[np.invert(wordData.isin(drops))]
    wordData = wordData.value_counts()
    wordData = wordData.where(wordData>limit).dropna()
    return wordData

def getGroup(value,groupDef):
    group = value
    
    for k,v in groupDef.items():
        if value in v:
            group = k
    return group

def figData_IdeaGroups(ideaList,ideaGroups,limit=0):
    ideaList = ideaList.map(lambda x: getGroup(x,ideaGroups)).value_counts()
    ideaList = ideaList.where(ideaList>limit).dropna()
    return ideaList

def figData_KeywordMatch(ideaList,keywordList):
    s = pd.Series({ k:ideaList.str.contains(k).sum() for k in keywordList})
    return s

def showVbar(title, figData, sortDir = 0, plotHeight = 350):
    keys = list(figData.keys())
    values = list(figData)
    sorted_keys = sorted(keys, key=lambda x: values[keys.index(x)])
    if isinstance(keys[0],str):
        if sortDir > 0:
            keyOrder = sorted_keys
        elif sortDir < 0:
            keyOrder = list(reversed(sorted_keys))
        else:
            keyOrder = list(sorted(keys))
        p = bkp.figure(x_range=keyOrder, plot_height=plotHeight, title=title, toolbar_location=None, tools="")
    else:
        p = bkp.figure(plot_height=350, title=title, toolbar_location=None, tools="")  
        
    p.vbar(x=keys, top=values, width=0.9, color=colourPrimary)

    p.xgrid.grid_line_color = None
    p.y_range.start = 0

    bkp.show(p)

def showHbar(title, figData, sortDir = 0, plotHeight = 350):
    keys = list(figData.keys())
    values = list(figData)
    sorted_keys = sorted(keys, key=lambda x: values[keys.index(x)])
    if isinstance(keys[0],str):
        if sortDir > 0:
            keyOrder = sorted_keys
        elif sortDir < 0:
            keyOrder = list(reversed(sorted_keys))
        else:
            keyOrder = list(sorted(keys))
        p = bkp.figure(y_range=keyOrder, plot_height=plotHeight, title=title, toolbar_location=None, tools="")
    else:
        p = bkp.figure(plot_height=350, title=title, toolbar_location=None, tools="")  
        
    p.hbar(y=keys, right=values, height=0.7, color=colourPrimary)

    p.ygrid.grid_line_color = None
    p.x_range.start = 0

    bkp.show(p)

collectors = {
    "251581315":{'name':"Members",'sent':91},
    "251581382":{'name':"Attenders",'sent':133}
}

collectorCounts = df.idCollector.value_counts()
collectorCounts = collectorCounts[collectorCounts.index.isin(list(collectors.keys()))]
dfCollectors = pd.DataFrame.from_dict(collectors,'index')
dfCollectors['response'] = collectorCounts
dfCollectors['silence'] = dfCollectors.sent - dfCollectors.response
dfCollectors = dfCollectors.append(dfCollectors.sum(numeric_only=True).append(pd.Series({'name':'All'})), ignore_index=True)

def showSurveyResponseRate():
    collectors = dfCollectors.name
    activity = ['response','silent']
    colors = [colourPrimary, "lightgray"]

    data = {'collectors' : collectors,
            'response'   : dfCollectors.response,
            'silent'   : dfCollectors.silence}

    p = bkp.figure(y_range=collectors, plot_height=250, title="survey response rate",
               toolbar_location=None, tools="")

    p.hbar_stack(activity, y='collectors', height=0.9, color=colors, source=data,
                 legend_label=activity)

    p.x_range.start = 0
    p.y_range.range_padding = 0.1
    p.ygrid.grid_line_color = None
    p.axis.minor_tick_line_color = None
    p.outline_line_color = None
    p.legend.location = "bottom_right"
    p.legend.orientation = "horizontal"

    bkp.show(p)
